webservers
web1, web2
dbservers
db1
inventory/group_vars/webservers.yml → webserver_port: 80, app_env: production
@webservers:
|--web1
|--web2
@dbservers:
|--db1 
[webservers]
web1 ansible_host=10.0.1.10 ansible_user=ubuntu
web2 ansible_host=10.0.1.11 ansible_user=ubuntu

[dbservers]
db1 ansible_host=10.0.1.20 ansible_user=ubuntu
all:
  children:
    webservers:
      hosts:
        web1:
          ansible_host: 10.0.1.10
          ansible_user: ubuntu
