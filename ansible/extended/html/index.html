Quick start commands
cd ~/terraform-ansible-labs/ansible/extended/labs
# Edit inventory/hosts.ini with your private IPs
ansible-inventory -i inventory/hosts.ini --graph
ansible -i inventory/hosts.ini webservers -m ansible.builtin.ping
# Expect: "ping": "pong" on each host
ansible webservers -i inventory/hosts.ini \
  -m ansible.builtin.setup -a "filter=ansible_distribution*"
ansible-playbook --syntax-check playbooks/loops-packages.yml
ansible-playbook -i inventory/hosts.ini playbooks/loops-packages.yml
export AWS_PROFILE=lab-terraform
ansible-inventory -i inventory/aws_ec2.yml --graph
ansible -i inventory/aws_ec2.yml role_webservers -m ansible.builtin.ping
